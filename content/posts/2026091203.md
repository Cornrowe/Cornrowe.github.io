import okhttp3.*;
import com.google.gson.*;

public class ClaudeOpusExample {
    public static void main(String[] args) throws Exception {
        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(60, java.util.concurrent.TimeUnit.SECONDS)
                .readTimeout(120, java.util.concurrent.TimeUnit.SECONDS)
                .build();

        // 替换为你自己申请的API Key

        // 构建请求体
        JsonObject message = new JsonObject();
        message.addProperty("role", "user");
        message.addProperty("content", "请用中文详细介绍一下云计算的历史。");

        JsonArray messages = new JsonArray();
        messages.add(message);

        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", "claude-3-opus-20240229");
        requestBody.add("messages", messages);
        requestBody.addProperty("max_tokens", 1000);
        requestBody.addProperty("temperature", 0.7);

        // 注意：这里的URL是关键的改变点！
        // 原来如果是官方，这里可能是 https://api.anthropic.com/v1/messages

        Request request = new Request.Builder()
                .url(url)
                .addHeader("Authorization", "Bearer " + apiKey)
                .post(RequestBody.create(requestBody.toString(), JSON))
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful()) {
                String responseData = response.body().string();
                System.out.println("调用成功！返回内容：\n" + responseData);
            } else {
                System.out.println("请求失败，状态码：" + response.code());
                System.out.println("错误信息：" + response.body().string());
            }
        }
    }
}
