import java.net.*;
import java.io.*;
import javax.json.*;

public class MoonshotTest {
    public static void main(String[] args) throws Exception {
        URL url = new URL("https://www.yunwuai.cc/v1/chat/completions");
        HttpURLConnection conn = (HttpURLConnection) url.openConnection();
        conn.setRequestMethod("POST");
        conn.setRequestProperty("Content-Type", "application/json");
        conn.setRequestProperty("Authorization", "Bearer " + "你的云雾APIKey");
        conn.setDoOutput(true);

        // 构建请求体
        String inputJson = "{"
            + "\"model\": \"moonshot-v1\","
            + "\"messages\": [{\"role\": \"user\", \"content\": \"用Java写一个Hello World\"}],"
            + "\"stream\": false"
            + "}";

        try(OutputStream os = conn.getOutputStream()) {
            byte[] input = inputJson.getBytes("utf-8");
            os.write(input, 0, input.length);
        }

        // 读取响应
        try(BufferedReader br = new BufferedReader(
                new InputStreamReader(conn.getInputStream(), "utf-8"))) {
            StringBuilder response = new StringBuilder();
            String responseLine = null;
            while ((responseLine = br.readLine()) != null) {
                response.append(responseLine.trim());
            }
            // 这里你应该解析JSON，拿choices[0].message.content
            System.out.println(response.toString());
        }
    }
}
