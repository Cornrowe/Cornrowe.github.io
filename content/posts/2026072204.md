### Java 实战：5分钟完成Claude Opus 4.1调用

我们直接进入最硬核的环节。以下代码基于纯Java标准库 + OkHttp，任何Java项目都适用。本例子演示 **流式输出(Streaming)**，因为这才是对话应用的标配——像ChatGPT打字一样，一个字一个字往外蹦，体验最好。

首先，在你的`pom.xml`里加入这个依赖：

xml
<dependency>
    <groupId>com.squareup.okhttp3</groupId>
    <artifactId>okhttp</artifactId>
    <version>4.12.0</version>
</dependency>


然后，直接上代码：

java
import okhttp3.*;
import org.json.JSONArray;
import org.json.JSONObject;

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.concurrent.TimeUnit;

public class ClaudeOpus41StreamClient {

    // 核心：[云雾ai大模型中转站](https://www.yunwuai.cc/)提供的API端点
    private static final String BASE_URL = "https://www.yunwuai.cc/v1";
    // 你的API密钥
    private static final String API_KEY = "你刚才复制的密钥";

    public static void main(String[] args) throws Exception {
        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(60, TimeUnit.SECONDS)
                .readTimeout(60, TimeUnit.SECONDS)
                .build();

        // 构建请求体 —— 使用Claude Opus 4.1模型
        JSONObject requestBody = new JSONObject();
        requestBody.put("model", "claude-3-opus-4.1");
        // 流式输出必须为true
        requestBody.put("stream", true);
        requestBody.put("max_tokens", 4096);
        // 构建消息数组
        JSONArray messages = new JSONArray();
        // System Prompt：身份设定
        messages.put(new JSONObject()
                .put("role", "system")
                .put("content", "你是一位资深的Java架构师，回答必须专业且专注解决问题。"));
        // User Prompt：用户提问
        messages.put(new JSONObject()
                .put("role", "user")
                .put("content", "请用Java代码示例，解释一下如何使用CompletableFuture实现异步任务编排。"));
        requestBody.put("messages", messages);

        // 构建请求
        Request request = new Request.Builder()
                .url(BASE_URL + "/chat/completions")
                .addHeader("Authorization", "Bearer " + API_KEY)
                .addHeader("Content-Type", "application/json")
                .post(RequestBody.create(MediaType.parse("application/json; charset=utf-8"), requestBody.toString()))
                .build();

        // 发送请求并处理流式响应
        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                System.out.println("请求失败! HTTP状态码: " + response.code());
                System.out.println("错误信息: " + (response.body() != null ? response.body().string() : "无"));
                return;
            }
            // 读取流数据
            BufferedReader reader = new BufferedReader(new InputStreamReader(response.body().byteStream(), "UTF-8"));
            String line;
            while ((line = reader.readLine()) != null) {
                // 流式SSE数据格式： data: {...
                if (line.startsWith("data: ")) {
                    String data = line.substring(6);
                    if ("[DONE]".equals(data)) {
                        break;
                    }
                    try {
                        JSONObject json = new JSONObject(data);
                        JSONArray choices = json.getJSONArray("choices");
                        if (choices.length() > 0) {
                            JSONObject delta = choices.getJSONObject(0).getJSONObject("delta");
                            String content = delta.optString("content", "");
                            // 打印出来，就像AI在打字
                            System.out.print(content);
                        }
                    } catch (Exception e) {
                        // 忽略格式异常数据
                    }
                }
            }
            System.out.println("\n\n--- 响应结束 ---");
        }
    }
}
