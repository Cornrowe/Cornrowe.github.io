### 第二步：用RestTemplate写一个o4-mini调用示例

这是最常见的Java写法，你用Spring Boot项目的话，直接复制下面这段代码，然后把`YOUR_API_KEY`替换成你的Key就行：

java
import org.springframework.web.client.RestTemplate;
import org.springframework.http.HttpEntity;
import org.springframework.http.HttpHeaders;
import org.springframework.http.HttpMethod;
import org.springframework.http.ResponseEntity;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.JsonNode;

import java.util.Map;
import java.util.List;

public class O4MiniExample {
    private static final String BASE_URL = "https://www.yunwuai.cc/v1";
    private static final String API_KEY = "sk-your-api-key-here"; // 替换你的Key

    public static String callO4Mini(String userMessage) {
        RestTemplate restTemplate = new RestTemplate();
        ObjectMapper mapper = new ObjectMapper();

        // 构建请求体
        Map<String, Object> requestBody = Map.of(
            "model", "o4-mini",
            "messages", List.of(
                Map.of("role", "user", "content", userMessage)
            )
        );

        // 设置请求头
        HttpHeaders headers = new HttpHeaders();
        headers.set("Authorization", "Bearer " + API_KEY);
        headers.set("Content-Type", "application/json");

        HttpEntity<String> entity = new HttpEntity<>(mapper.writeValueAsString(requestBody), headers);

        // 发送请求
        ResponseEntity<String> response = restTemplate.exchange(
            BASE_URL + "/chat/completions",
            HttpMethod.POST,
            entity,
            String.class
        );

        // 解析响应
        try {
            JsonNode root = mapper.readTree(response.getBody());
            return root.path("choices").get(0).path("message").path("content").asText();
        } catch (Exception e) {
            return "调用出错: " + e.getMessage();
        }
    }

    public static void main(String[] args) {
        String reply = callO4Mini("请用Java写一个单例模式的示例");
        System.out.println(reply);
    }
}


**注意看这里的关键改动：**

原来用OpenAI官方API时，你写的`baseUrl`是`https://api.openai.com/v1`。换成云雾后，你只需要把这一行改成：

### 第三步：用OkHttp写一个带流式输出的版本

如果你想用流式（streaming）输出，实时看到o4-mini的回复，用OkHttp也很简单：

java
import okhttp3.*;
import com.fasterxml.jackson.databind.ObjectMapper;

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.Map;
import java.util.List;

public class O4MiniStreamingExample {
    private static final String BASE_URL = "https://www.yunwuai.cc/v1";
    private static final String API_KEY = "sk-your-api-key-here";
    private static final OkHttpClient client = new OkHttpClient();
    private static final ObjectMapper mapper = new ObjectMapper();

    public static void streamO4Mini(String userMessage) {
        Map<String, Object> requestBody = Map.of(
            "model", "o4-mini",
            "messages", List.of(
                Map.of("role", "user", "content", userMessage)
            ),
            "stream", true
        );

        RequestBody body;
        try {
            body = RequestBody.create(
                mapper.writeValueAsString(requestBody),
                MediaType.parse("application/json")
            );
        } catch (Exception e) {
            System.out.println("构建请求体失败: " + e.getMessage());
            return;
        }

        Request request = new Request.Builder()
            .url(BASE_URL + "/chat/completions")
            .addHeader("Authorization", "Bearer " + API_KEY)
            .post(body)
            .build();

        // 流式读取响应
        client.newCall(request).enqueue(new Callback() {
            @Override
            public void onFailure(Call call, IOException e) {
                System.out.println("请求失败: " + e.getMessage());
            }

            @Override
            public void onResponse(Call call, Response response) throws IOException {
                if (!response.isSuccessful()) {
                    System.out.println("响应错误: " + response.code());
                    return;
                }
                BufferedReader reader = new BufferedReader(
                    new InputStreamReader(response.body().byteStream())
                );
                String line;
                while ((line = reader.readLine()) != null) {
                    if (line.startsWith("data: ") && !line.equals("data: [DONE]")) {
                        String jsonData = line.substring(6);
                        try {
                            String content = mapper.readTree(jsonData)
                                .path("choices")
                                .get(0)
                                .path("delta")
                                .path("content")
                                .asText();
                            if (content != null && !content.isEmpty()) {
                                System.out.print(content); // 实时打印结果
                            }
                        } catch (Exception e) {
                            // 忽略解析错误
                        }
                    }
                }
            }
        });
    }

    public static void main(String[] args) throws InterruptedException {
        streamO4Mini("给我写一个Java 8的Stream示例");
        Thread.sleep(30000); // 等待流式输出完成
    }
}
