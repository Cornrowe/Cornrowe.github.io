// 在微信小程序某个合适的地方调用此函数，比如一个按钮的点击事件

function callMultiModelAPI(userQuestion) {
  const apiKey = 'sk-你的密钥'; // *必改项：换成你从后台获取的API Key*
  const apiUrl = 'https://www.yunwuai.cc/v1/chat/completions'; // 完全兼容OpenAI的接口

  wx.request({
    url: apiUrl,
    method: 'POST',
    header: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${apiKey}`
    },
    data: {
      model: 'gpt-4o-mini', // 这里选哪个模型，就决定了回答的质量和成本
      messages: [
        {
          role: 'user',
          content: userQuestion
        }
      ],
      temperature: 0.7,
    },
    success(res) {
      // 成功拿到回复，处理返回数据
      const reply = res.data.choices[0].message.content;
      console.log('AI回复：', reply);
      // 这里可以把reply显示在小程序页面上
      // this.setData({ aiReply: reply });
    },
    fail(err) {
      // 请求失败，大概率是白名单或网络问题
      console.error('API调用失败：', err);
      // 这里可以弹个提示：wx.showToast({ title: '请求失败，请检查网络', icon: 'none' });
    }
  });
}

// 调用示例
callMultiModelAPI('请用一句话解释什么是量子纠缠？');

大概率是你选的模型问题。比如 `o1-preview` 这种深层推理模型，它本身响应就慢，不适合小程序里的实时对话场景。建议小程序里首选 `gpt-4o-mini` 或 `deepseek-chat`，响应通常在2-5秒内。云雾平台官网（www.yunwuai.cc）的模型支持列表里会有标注，有些模型适合高并发，不适合高并发时我就避开。

这个现象最可能的原因是微信小程序对返回数据的编码处理多了一层转义。你可以在 `success` 回调里先 `console.log(res)` 看看原始数据。如果数据里包着 `\"` 转义符，你可以用 `JSON.parse(reply)` 或 `reply.replace(/\\/g, '')` 简单修复一下。如果还没解决，试试把 `model` 改成支持原生Markdown格式的模型，比如 `claude-haiku`。
