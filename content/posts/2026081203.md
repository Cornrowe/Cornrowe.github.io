### 2. 核心代码：调用Qwen-VL

我写了个最简单的HTTP POST调用，模拟发送一张网络图片让模型识别。

java
import java.net.URI;
import java.net.http.*;
import java.net.http.HttpRequest.BodyPublishers;
import java.net.http.HttpResponse.BodyHandlers;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.node.*;

public class QwenVLClient {
    public static void main(String[] args) throws Exception {
        // 这里是[云雾ai大模型中转站](https://www.yunwuai.cc/)的API地址
        String baseUrl = "https://www.yunwuai.cc/v1";
        // 你自己的API Key
        String apiKey = "sk-your-api-key-here";

        // 构建请求体
        ObjectMapper mapper = new ObjectMapper();
        ObjectNode requestBody = mapper.createObjectNode();
        requestBody.put("model", "qwen-vl-plus"); // 具体模型名按文档选择

        ArrayNode messages = requestBody.putArray("messages");
        ObjectNode userMessage = messages.addObject();
        userMessage.put("role", "user");

        ArrayNode content = userMessage.putArray("content");
        ObjectNode textPart = content.addObject();
        textPart.put("type", "text");
        textPart.put("text", "请描述这张图片的内容");

        ObjectNode imagePart = content.addObject();
        imagePart.put("type", "image_url");
        ObjectNode imageUrl = imagePart.putObject("image_url");
        // 这里放你的图片URL，用一张测试图
        imageUrl.put("url", "https://www.yunwuai.cc/register?channel=c_7o7g8tlk");

        // 发送请求
        HttpClient client = HttpClient.newHttpClient();
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(baseUrl + "/chat/completions"))
                .header("Content-Type", "application/json")
                .header("Authorization", "Bearer " + apiKey)
                .POST(BodyPublishers.ofString(mapper.writeValueAsString(requestBody)))
                .build();

        HttpResponse<String> response = client.send(request, BodyHandlers.ofString());
        System.out.println(response.body());
    }
}


这段代码跑通后，你就能拿到一模一样的Qwen-VL图片理解结果，但成本完全不一样。
