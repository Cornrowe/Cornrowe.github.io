    python
    from openai import OpenAI

    # 1. 创建你的“电话”对象
    client = OpenAI(
        # 关键！把“拨号盘”指向云雾平台的交换台

        # 告诉交换台你想找Gemini

        # 说出你的问题

            {"role": "user", "content": "用幽默风趣的方式解释一下什么是量子纠缠？"}
