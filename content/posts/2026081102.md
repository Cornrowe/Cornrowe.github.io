  apiKey: '你的云雾api密钥', // 从云雾api后台复制
  baseURL: 'https://www.yunwuai.cc/v1',
});

async function main() {
  const stream = await openai.chat.completions.create({
    model: 'gpt-5', // 或其他你想用的模型，比如gpt-4o，claude-3-5-sonnet
    messages: [{ role: 'user', content: '写一个200字的关于云计算的小故事。' }],
    stream: true, // 流式输出，体验感拉满
  });

  for await (const chunk of stream) {
    process.stdout.write(chunk.choices[0]?.delta?.content || '');
  }
}

main();
