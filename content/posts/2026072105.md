### 第四步：创建API客户端并发送第一条消息

新建一个文件 `chat.js`，写入以下代码：

javascript
const OpenAI = require('openai');
require('dotenv').config();

const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
  baseURL: process.env.OPENAI_BASE_URL || 'https://www.yunwuai.cc/v1',
});

async function main() {
  try {
    const response = await client.chat.completions.create({
      model: 'gpt-4o', // 从下面第三个例子起，你可以换成 gpt-4、gpt-4o-mini、o1 等
      messages: [
        { role: 'system', content: '你是一位乐于助人的助手。' },
        { role: 'user', content: '用json格式告诉我怎么在Node.js里调用OpenAI API。' },
      ],
      stream: true, // 你很快会在流式响应示例里看到这个开关的巨大优势
    });

    for await (const chunk of response) {
      process.stdout.write(chunk.choices[0]?.delta?.content || '');
    }
  } catch (error) {
    console.error('调用云端API时出错：', error.message);
  }
}

main();


这段代码做了什么？
- 用环境变量初始化客户端。
- 调用 `chat.completions.create` 向 `gpt-4o` 模型发送一条消息。
- `stream: true` 开启了流式响应——你会看到模型一个字一个字地输出，而不必等整个回复生成完。这对实时应用（如智能客服、聊天机器人）非常关键。
- `for await ... of` 语法逐块读取流式响应，并直接打印到控制台。

运行代码：

几秒钟后，你应该看到模型开始输出JSON格式的Node.js调用指南。**只要能看到这个流式输出，就说明你的Node.js项目已经成功接入了GPT-4，并且全程没有走任何代理。**

如果你希望以非流式方式一次性等待完整回复，只需把 `stream: true` 删除或改为 `false`，并直接访问 `response.choices[0].message.content`。

## 让你的Node.js应用更健壮：处理错误与重试

接入API只是第一步，生产环境中你需要处理网络波动和模型限流。下面是一段增加自动重试机制和超时控制的代码，可以直接套用你的项目里：

javascript
const OpenAI = require('openai');
require('dotenv').config();

const retryDemo = async () => {
  const openai = new OpenAI({
    apiKey: process.env.OPENAI_API_KEY,
    baseURL: 'https://www.yunwuai.cc/v1',
    timeout: 10000, // 10秒超时
    maxRetries: 3,   // 默认重试3次
  });

  for (let attempt = 1; attempt <= 3; attempt++) {
    try {
      const comp = await openai.chat.completions.create({
        model: 'gpt-4o-mini',
        messages: [{ role: 'user', content: 'ping' }],
      });
      console.log('成功第'+attempt+'次尝试：', comp.choices[0].message.content);
      break;
    } catch (err) {
      console.error(`尝试 ${attempt}/3 失败：${err.message}`);
      if (attempt === 3) throw err;
    }
  }
};

retryDemo();


把这段代码加到你的工具函数里，即使网络偶尔抖一下，你的应用也能自动恢复，用户几乎感觉不到异常。
