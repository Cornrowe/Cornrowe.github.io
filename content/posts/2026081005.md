### 复制即用的完整代码示例

java
import okhttp3.*;
import com.google.gson.JsonObject;
import java.io.IOException;

public class ClaudeHaikuDemo {
    public static void main(String[] args) {
        // 1. 配置核心参数
        String apiKey = "在这里填写你在[云雾api聚合平台](https://www.yunwuai.cc/)申请的API Key";
        String apiUrl = "https://www.yunwuai.cc/v1/chat/completions";
        String modelName = "claude-3-haiku-20240307"; // 指定使用Claude Haiku模型

        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(30, java.util.concurrent.TimeUnit.SECONDS)
                .writeTimeout(60, java.util.concurrent.TimeUnit.SECONDS)
                .readTimeout(60, java.util.concurrent.TimeUnit.SECONDS)
                .build();

        // 2. 构建请求体（完全兼容OpenAI格式）
        MediaType mediaType = MediaType.get("application/json; charset=utf-8");
        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", modelName);

        JsonObject message = new JsonObject();
        message.addProperty("role", "user");
        message.addProperty("content", "用Java写一个简单的Hello World程序，并解释每一行的作用。");
        requestBody.add("messages", new com.google.gson.JsonArray());
        requestBody.getAsJsonArray("messages").add(message);
        requestBody.addProperty("temperature", 0.7); // 控制创造性
        requestBody.addProperty("max_tokens", 500);  // 最大回复长度

        Request request = new Request.Builder()
                .url(apiUrl)
                .addHeader("Authorization", "Bearer " + apiKey)
                .addHeader("Content-Type", "application/json")
                .post(RequestBody.create(requestBody.toString(), mediaType))
                .build();

        // 3. 发起同步请求并打印结果
        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful() && response.body() != null) {
                String responseBody = response.body().string();
                System.out.println("Claude Haiku的回复：");
                System.out.println(responseBody);
            } else {
                System.err.println("请求失败，状态码: " + response.code());
                System.err.println("错误信息: " + response.body().string());
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
