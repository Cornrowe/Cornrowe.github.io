# 正确配置：OpenAI兼容格式

    model="deepseek-v3",
    messages=[
        {"role": "system", "content": "你是一个有用的助手。"},
        {"role": "user", "content": "请用100字介绍深度求索公司。"}
    ],
    stream=True,  # 启用流式输出
    max_tokens=512

    print(chunk.choices[0].delta.get("content", ""), end="")
