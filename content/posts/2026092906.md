    # 模型名称：注意这里是claude-3-5-sonnet-20241022 (云雾中转站专有映射)

**第三步：编写Java代码**

java
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import com.theokanning.openai.service.OpenAiService;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;

import java.time.Duration;
import java.util.ArrayList;
import java.util.List;

@Component
public class ClaudeSonnetService {

    @Value("${ai.api.base-url}")
    private String baseUrl;

    @Value("${ai.api.api-key}")
    private String apiKey;

    @Value("${ai.api.model}")
    private String model;

    public String callClaudeSonnet(String userMessage) {
        // 1. 构建OpenAI服务（实际上调用的是Claude模型）
        OpenAiService service = new OpenAiService(apiKey, Duration.ofSeconds(60));
        // 关键一步：修改API基地址
        service.setBaseUrl(baseUrl);

        // 2. 构建消息列表
        List<ChatMessage> messages = new ArrayList<>();
        messages.add(new ChatMessage("system", "You are a helpful Java assistant."));
        messages.add(new ChatMessage("user", userMessage));

        // 3. 构建请求
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model(model) // 模型名称
                .messages(messages)
                .maxTokens(2000)
                .temperature(0.7)
                .build();

        // 4. 发起调用
        try {
            ChatCompletionResponse response = service.createChatCompletion(request);
            // 返回结果
            return response.getChoices().get(0).getMessage().getContent();
        } catch (Exception e) {
            e.printStackTrace();
            return "调用失败: " + e.getMessage();
        } finally {
            service.shutdownExecutor(); // 优雅关闭
        }
    }
}
