import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import java.time.Duration;
import java.util.Arrays;

public class QwenPlusDemo {
    public static void main(String[] args) {
        // 1. 设置[云雾ai聚合站](https://www.yunwuai.cc/)的API密钥和自定义端点
        String token = "sk-your-cloud-api-key"; // 从云雾后台获取
        String baseUrl = "https://www.yunwuai.cc/v1";

        // 2. 创建服务实例，指定超时和自定义URL
        OpenAiService service = new OpenAiService(token, Duration.ofSeconds(30));
        // 注意：现代版本的OpenAiService构造函数支持传入自定义的HttpClient和URL
        // 或者你可以使用OkHttpClient来手动构造，但最简单的做法是：
        System.setProperty("openai.base.url", baseUrl);

        // 3. 构建请求，model参数写“qwen-plus”即可
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("qwen-plus") // 关键：模型名
                .messages(Arrays.asList(
                        new ChatMessage("system", "你是一位资深Java架构师。"),
                        new ChatMessage("user", "请用Java写一个线程安全的单例模式。")
                ))
                .maxTokens(2048)
                .build();

        // 4. 执行并输出
        service.createChatCompletion(request)
                .getChoices()
                .forEach(choice -> System.out.println(choice.getMessage().getContent()));

        // 5. 关闭服务
        service.shutdownExecutor();
    }
}
