### 2. Java代码示例

以下Java代码演示了如何调用 `gemini-2.5-pro` 模型：

java
import okhttp3.*;
import com.google.gson.JsonArray;
import com.google.gson.JsonObject;

import java.io.IOException;

public class GeminiApiCall {
    public static void main(String[] args) {
        // 这里你只需要做两件事：
        // 1. 更换API密钥
        // 2. 修改Base URL
        String apiKey = "sk-你从[云雾ai聚合平台](https://www.yunwuai.cc/)申请的Key"; // 关键点1：使用云雾的Key
        // 这是与OpenAI兼容的接口，支持生成和聊天
        String url = "https://www.yunwuai.cc/v1/chat/completions"; // 关键点2：修改Base URL

        OkHttpClient client = new OkHttpClient().newBuilder()
                .build();

        // 构建请求体
        JsonObject requestBody = new JsonObject();
        // 这里可以换成任何Gemini模型：gemini-2.5-pro, gemini-2.5-flash 等
        requestBody.addProperty("model", "gemini-2.5-pro");

        JsonArray messages = new JsonArray();
        JsonObject userMessage = new JsonObject();
        userMessage.addProperty("role", "user");
        userMessage.addProperty("content", "用中文解释一下什么是RAG（检索增强生成）技术，简洁一点。");
        messages.add(userMessage);

        requestBody.add("messages", messages);
        requestBody.addProperty("temperature", 0.7);
        requestBody.addProperty("max_tokens", 1000);

        // 构建请求
        Request request = new Request.Builder()
                .url(url)
                .addHeader("Authorization", "Bearer " + apiKey)
                .post(RequestBody.create(
                        requestBody.toString(),
                        MediaType.get("application/json; charset=utf-8")
                ))
                .build();

        // 发送请求并获取响应
        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful() && response.body() != null) {
                String responseBody = response.body().string();
                System.out.println("AI返回结果：");
                System.out.println(responseBody);

                // 简单解析返回的content
                JsonObject jsonResponse = new com.google.gson.JsonParser().parseString(responseBody).getAsJsonObject();
                String content = jsonResponse.getAsJsonArray("choices")
                        .get(0).getAsJsonObject()
                        .getAsJsonObject("message")
                        .get("content").getAsString();
                System.out.println("提取后的文本：" + content);
            } else {
                System.out.println("请求失败，状态码：" + response.code());
                if (response.body() != null) {
                    System.out.println("错误信息：" + response.body().string());
                }
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
