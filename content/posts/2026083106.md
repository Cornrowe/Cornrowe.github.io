### 完整代码

javascript
// app.js
import OpenAI from 'openai';

// 配置[云雾api中转站](https://www.yunwuai.cc/)
const client = new OpenAI({
  apiKey: '你的云雾API Key', // 替换为你的实际Key
  baseURL: 'https://www.yunwuai.cc/v1', // 关键：改成云雾直连地址
});

// 聊天完成示例
async function chatWithGPT() {
  try {
    const completion = await client.chat.completions.create({
      model: 'gpt-4o', // 或其他支持的模型名
      messages: [
        { role: 'system', content: '你是一个专业的Node.js助手。' },
        { role: 'user', content: '请用Node.js写一个简单的HTTP服务器。' },
      ],
      temperature: 0.7,
      max_tokens: 500,
    });

    console.log('回复内容:', completion.choices[0]?.message?.content);
  } catch (error) {
    console.error('调用失败:', error);
  }
}

chatWithGPT();
